%%===============================================================%
%%                  Macros to append papers                      %

\newcounter{papercounter}
\setcounter{papercounter}{0}

\newcommand{\paperpage}[4]{
\newpage
\refstepcounter{papercounter}
\renewcommand{\thepage}{}
\thispagestyle{empty}
\cleardoublepage
\hfill{\Large \textbf{Paper \romannum{\thepapercounter}}}
\label{app:paper_\arabic{papercounter}}
\vspace*{\fill}
    \begin{center}
        \textbf{\Large #2}\\
		\vspace{0.5cm}
        #1\\
		\vspace{0.5cm}
        #3
    \end{center}
\vfill
#4
\addcontentsline{toc}{chapter}{\textbf{Paper \romannum{\thepapercounter}} - #2}
%----- include -----
\newpage
\thispagestyle{empty}
\cleardoublepage
}

% To avoid adding things to the TOC
\makeatletter

\newcommand{\nocontentsline}[3]{}
\newcommand{\tocless}[2]{\bgroup\let\addcontentsline=\nocontentsline#1{#2}\egroup}

\let\oldsection\section
\newcommand{\@sectionstar}[1]{{\oldsection*{#1}}}
\newcommand{\@sectionnostar}[1]{{\tocless\oldsection{#1}}}
\renewcommand{\section}{\@ifstar{\@sectionstar}{\@sectionnostar}}


\let\oldsubsection\subsection
\newcommand{\@subsectionstar}[1]{{\oldsubsection*{#1}}}
\newcommand{\@subsectionnostar}[1]{{\tocless\oldsubsection{#1}}}
\renewcommand{\subsection}{\@ifstar{\@subsectionstar}{\@subsectionnostar}}

\makeatother
