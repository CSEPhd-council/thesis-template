%%===============================================================%
%%                  Macros to append papers                      %
%%===============================================================%
\newcounter{papercounter}
\setcounter{papercounter}{0}
\renewcommand{\thepapercounter}{\romannum{\value{papercounter}}} % Change it so that \thepapercounter is displayed as a roman numeral

\newcommand{\paperpage}[4]{
\newpage
\refstepcounter{chapter} % Advance chapter count for separate href targets in papers
\refstepcounter{papercounter}
\addcontentsline{toc}{chapter}{\textbf{Paper \thepapercounter} - #2}
\renewcommand{\thepage}{}
\thispagestyle{empty}
\cleardoublepage
\hfill{\Large \textbf{Paper \thepapercounter}}
\vspace*{\fill}
    \begin{center}
        \textbf{\Large #2}\\
		\vspace{0.5cm}
        #1\\
		\vspace{0.5cm}
        #3
    \end{center}
\vfill
#4
%----- include -----
\newpage
\thispagestyle{empty}
\cleardoublepage
}

% To avoid adding sections (and lower) to the TOC. Set to eg. 1 to show sections but not sub-sections
\addtocontents{toc}{\protect\setcounter{tocdepth}{0}}

% Hide chapter number in appended paper sections, as the papers can be seen as appended and not part of the chapters in the kappa
\counterwithout{section}{chapter}
\counterwithout{figure}{chapter}
\counterwithout{table}{chapter}
